#include <tunables/global>

profile ha-cops flags=(attach_disconnected,mediate_deleted,complain) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities
  capability chown,
  capability setuid,
  capability setgid,
  capability sys_resource,
  capability dac_override,
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for service
  /usr/sbin/nginx cx -> nginx,

  profile nginx flags=(attach_disconnected,mediate_deleted,complain) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    capability net_bind_service,
    network inet stream,
    network inet6 stream,

    # Core Nginx files and configuration
    /usr/sbin/nginx r,
    /etc/nginx/** r,

    # Access to options.json and other files within your addon
    /data/** rw,

    # Access to mapped volumes specified in config.json
    /share/** r,
    /media/** r,

    # Access to the web application files Nginx is serving
    /cops/** r,

    # Access required for service functionality
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,

    # Nginx operational runtime directories (Crucial for startup)
    /var/lib/nginx/{,**} rwk,
    /var/log/nginx/{,**} rw,
    /run/nginx[._]* rwk,
    /run/nginx.pid rwk,

    # Nginx Temporary/Buffer Directories (Prevents 500 errors under load)
    /var/tmp/nginx/{,**} rwk,
    /tmp/nginx/{,**} rwk,
  }


}
