#include <tunables/global>

profile ha-cops flags=(complain,attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,
  /media/** rw,

  # Start new profile for service
  /usr/bin/php cx -> php,
  /usr/bin/rsync cx -> rsync,

  profile php {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ha-cops,

    # Read only access to the directory where library is stored
    /media/books/** r,

    # Access required for service functionality
    /usr/bin/php r,
  }

  profile rsync {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ha-cops,

    # ReadWrite access to directory where library is stored
    /media/books/** r,

    # Access required for service functionality
    /usr/bin/rsync r,
  }
}
