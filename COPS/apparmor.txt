#include <tunables/global>

profile ha-cops flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Capabilities
  capability chown,
  capability setuid,
  capability setgid,
  capability sys_resource,
  capability dac_override,
  capability kill,
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for service
  /usr/sbin/nginx cx -> nginx,
  /usr/sbin/php-fpm cx -> phpfpm,
  /usr/bin/rsync cx -> rsync,
  /usr/bin/korrosync cx -> korrosync,

  profile nginx flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ha-cops,

    capability chown,
    capability setuid,
    capability setgid,
    capability net_bind_service,
    capability dac_override,
    network inet stream,
    network inet6 stream,

    # Core Nginx files and configuration
    /usr/sbin/nginx r,
    /etc/nginx/** r,

    # Access to options.json and other files within your addon
    /data/** rw,

    # Access to mapped volumes specified in config.json
    /share/** r,
    /media/** r,

    # Access to the web application files Nginx is serving
    /cops/** r,

    # Access required for service functionality
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,

    # Nginx operational runtime directories (Crucial for startup)
    /var/lib/nginx/{,**} rwk,
    /var/log/nginx/{,**} rw,
    /run/nginx[._]* rwk,
    /run/nginx.pid rwk,

    # Nginx Temporary/Buffer Directories (Prevents 500 errors under load)
    /var/tmp/nginx/{,**} rwk,
    /tmp/nginx/{,**} rwk,

    # PID file and Logging access
    /run/nginx/nginx.pid rw,
    /run/nginx/nginx.pid w,
    /proc/*/fd/[0-9]* rw,
    /etc/ssl/openssl.cnf r,
  }

  profile phpfpm flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ha-cops,

    capability kill,

    # Core PHP-FPM binary and configuration read access
    /usr/sbin/php-fpm r,
    /etc/php*/** r,

    # Access required for basic execution framework
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,

    # Access to options.json and persistent add-on databases
    /data/** rw,

    # Access to the web application files (PHP needs read/write for caching/updates)
    /cops/** rw,

    # Access to user mapped volumes (Read/Write if COPS edits metadata/covers)
    /share/** r,
    /media/** r,

    # PHP session storage and temporary execution directories
    /var/lib/php/sessions/{,**} rwk,
    /tmp/{,**} rwk,

    # Communication bridge back to Nginx (Unix Domain Socket path)
    # If your S6 script puts the socket in /run/nginx/ or /tmp/:
    /run/nginx[._]* rwk,
    /tmp/nginx/{,**} rwk,
  }

  profile rsync flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ha-cops,

    capability chown,
    capability setuid,
    capability setgid,
    capability fowner,
    capability fsetid,
    capability dac_override,
    network inet stream,

    /usr/bin/rsync rm,
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,
    /etc/rsyncd.conf r,
    /run/rsyncd.pid rwk,

    # Allow rsync to access and interact with any container process file descriptors
    /proc/[0-9]*/fd/[0-9]* rw,

    # Dynamic user-configured destination path inside /media
    /media/ r,
    /media/*/{,**} rwk,

    # Access to options.json and persistent add-on databases
    /data/** rw,
  }

  profile korrosync flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_ha-cops,

    network inet stream,

    /usr/bin/korrosync rm,
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,

    # Access to options.json and persistent add-on databases
    /data/** rwk,

    # Allow reading process cgroup membership info
    /proc/[0-9]*/cgroup r,

    # Allow reading CPU allocations/limits from the cgroup filesystem
    /sys/fs/cgroup/cpu.max r,

    /data/korrosync/** rwk,
  }

}
